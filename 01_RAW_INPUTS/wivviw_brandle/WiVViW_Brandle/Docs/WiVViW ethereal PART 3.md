// lib/reportWriter.ts  
import { ChapterReport, Candidate } from "./types";

export function applyEventToReport(  
  report: ChapterReport,  
  event: any,  
  candidates: Candidate\[\]  
): ChapterReport {  
  const updated \= { ...report, updatedAt: new Date().toISOString() };

  if (event.type \=== "CONTEXT\_CONFIRMED") {  
    updated.confirmedContext \= { ...updated.confirmedContext, ...event.payload };  
  }

  if (event.type \=== "CONSTRAINT\_STATED") {  
    if (\!updated.constraints.includes(event.payload)) updated.constraints.push(event.payload);  
  }

  if (event.type \=== "CANDIDATE\_STATUS\_CHANGED") {  
    const { name, status, reason } \= event.payload;

    // rebuild lists from candidates to avoid drift  
    const promising \= candidates.filter(c \=\> c.status \=== "promising").map(c \=\> c.name);  
    const shortlisted \= candidates.filter(c \=\> c.status \=== "shortlisted").map(c \=\> c.name);

    updated.namesExplored.promising \= promising;  
    updated.namesExplored.shortlisted \= shortlisted;

    if (status \=== "blocked") {  
      updated.namesExplored.blocked \= \[  
        ...updated.namesExplored.blocked.filter(x \=\> x.name \!== name),  
        { name, reason: reason || "Blocked signal detected" }  
      \];  
    }

    if (status \=== "risky") {  
      updated.namesExplored.risky \= \[  
        ...updated.namesExplored.risky.filter(x \=\> x.name \!== name),  
        { name, reason: reason || "Risk signal detected" }  
      \];  
    }  
  }

  if (event.type \=== "SELECT\_FINAL") {  
    updated.namesExplored.selected \= event.payload.name;  
    updated.decisionsLocked.push(\`Selected name: ${event.payload.name}\`);  
  }

  return updated;  
}
